plot_calib_surv <- function(pi, y, tte, th, n_bins=10)
{
  #Sort the data
  n <- length(pi)
  o <- order(pi)
  pi <- pi[o]
  y <- y[o]
  tte <- tte[o]
  bin <- ntile(pi, 10) #TODO: can be made faster given the data are sorted
  
  out <- matrix(NA, nrow=n_bins, ncol=4)
  colnames(out) <- c("Predicted", "Observed", "CI_low", "CI_high")
  
  plot(NULL, xlim=c(0,1), ylim=c(0,1), xlab="Predicted", ylab="Observed")
 # Add the dynamic title to the plot
  title(paste("Calibration Plot at the End of", th / 365, "Year Follow-up"))
  
  lines(c(0,1),c(0,1), col='gray')
  
  for(i in 1:n_bins)
  {
    dt <- data.frame(pi=pi[which(bin==i)], y=y[which(bin==i)], tte=tte[which(bin==i)])
    km <- survfit(Surv(tte, y) ~ 1, data = dt)
    index <- min(which(km$time>=th))
    out[i,1] <- mean(dt$pi)
    out[i,2] <- 1-km$surv[index]
    out[i,3] <- 1-km$lower[index]
    out[i,4] <- 1-km$upper[index]
    
    lines(c(out[i,1],out[i,1]),c(out[i,3],out[i,4]))
  }
  
  lines(out[,1],out[,2])
}

5 Calibration plot Intercept and the Slop for 1 Year Follow up

 #pi: predicted risk at time horizon (th) of interest. 
  #y: status at th of interest
  #tte: time variable
  #bh: predicted baseilne hazard at th of interest
  #th: time horizon of interest
  stat_calib_surv <- function(pi, y, tte, bh, th)
  {
    #Calibration in the large
    km <- survfit(Surv(time=tte, event=y)~1)
    obs <- 1-km$surv[min(which(km$time>=th))]
    pred <- mean(pi)
    lin <- log(-log(1-pi))
    cph <- coxph(Surv(tte, y)~lin)
    slope <- unname(coefficients(cph))
    
    c(intercept=(pred-obs), slope=slope)
  }

inte_slop_1year<-stat_calib_surv(simpledata$pi1, simpledata$CNSR, simpledata$HOSPFU, bh365, th=365)
